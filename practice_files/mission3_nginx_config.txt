# This is a sample nginx config for Mission 3.
# We need to migrate the server name.

server {
    listen 80;
    server_name old-app.com www.old-app.com;

    location / {
        root /var/www/old-app.com/html;
        index index.html index.htm;
    }

    location /api {
        proxy_pass http://api.old-app.com;
    }
}
